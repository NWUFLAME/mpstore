# 页面级Store接入

页面Store的数据只能在当前页面及其子组件中使用

# 接入

# 原有初始化方式

Page({
  onLoad() {
    // 业务逻辑...
  },
})

# 接入方式(Page的原有参数填写到mpstore.Page的第二个参数即可)

import mpstore from '../../utils/mpstore'
import store from './store'

mpstore.Page(store, {
    // 当全局store配置了updateAll为false的时候，若页面要访问全局状态，需要在use中声明需要的字段，否则可以省略use配置项
  use: ['xxx'],
  onLoad() {
    // 业务逻辑...
  },
})

# 准备页面级Store

在页面js的同级目录创建store.js, API基本同Vuex一致

export default {
  state: {
      foo : 'foo',
      bar : 'bar'
      ...
  },
  getters: {
    // getters是响应式数据，所依赖的state改变或getters改变会自动重新计算，适合如购物车计算总价的场景
    test({ state, globalState, getters, globalGetters }) {
      return xxx
    },
  },
  actions: {
    // 第一个参数提供了context上下文环境，可以按需解构
    test({ state, globalState, dispatch, dispatchGlobal, commit, commitGlobal, getters, globalGetters }, payload) {
        // 业务逻辑... 一般是commit一个或多个mutation
    },

  },
  mutations: {
    test(state, payload) {
      // 业务逻辑... 一般是根据payload修改state的值, state.xxx = xxx
    },

  },
  
  // 是否开启控制台日志(打印store变化，类似Vuex-devtools/Redux-devtools)
  logger: false 
}
